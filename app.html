from flask import Flask, request, jsonify, render_template_string from datetime import datetime app = Flask(__name__) latest_report = None breadcrumbs = [] HTML_PAGE = """ BEAST Scout View

BEAST Scout

Ready
Tracking
Emergency
No active tracking yet.
""" @app.route("/") def scout_view(): return render_template_string(HTML_PAGE) @app.route("/report", methods=["POST"]) def receive_report(): global latest_report, breadcrumbs data = request.get_json() if not data: return jsonify({"status": "error", "message": "No data received"}), 400 latest_report = { "received_at": datetime.now().isoformat(), "type": data.get("type"), "point": data.get("point"), "breadcrumb_count": data.get("breadcrumb_count") } point = data.get("point") if point: breadcrumbs.append(point) print("NEW SCOUT REPORT") print(latest_report) return jsonify({"status": "received"}) @app.route("/agency") def agency_view(): return jsonify({ "latest_report": latest_report, "breadcrumbs": breadcrumbs }) if __name__ == "__main__": app.run(debug=True, host="0.0.0.0", port=5000)
